import random
import asyncio
import time
import aiosqlite
import discord
from discord.ext import commands
from discord import app_commands

DB = "bot.db"


class GiveawayButton(discord.ui.View):
    def __init__(self, giveaway_id):
        super().__init__(timeout=None)
        self.giveaway_id = giveaway_id

    @discord.ui.button(
        label="🎉 Teilnehmen",
        style=discord.ButtonStyle.green,
        custom_id="giveaway_join"
    )
    async def join(self, interaction: discord.Interaction, button: discord.ui.Button):

        async with aiosqlite.connect(DB) as db:

            try:
                await db.execute(
                    """
                    INSERT INTO giveaway_entries
                    VALUES (?,?)
                    """,
                    (
                        self.giveaway_id,
                        interaction.user.id
                    )
                )

                await db.commit()

                await interaction.response.send_message(
                    "✅ Du nimmst jetzt teil.",
                    ephemeral=True
                )

            except:

                await interaction.response.send_message(
                    "Du bist bereits eingetragen.",
                    ephemeral=True
                )


class Giveaway(commands.Cog):

    def __init__(self, bot):
        self.bot = bot
        self.bot.loop.create_task(self.check_giveaways())

    async def check_giveaways(self):

        await self.bot.wait_until_ready()

        while True:

            now = int(time.time())

            async with aiosqlite.connect(DB) as db:

                cur = await db.execute("""
                SELECT id,message_id,channel_id,winners,prize
                FROM giveaways
                WHERE ended=0
                AND end_time<=?
                """,(now,))

                giveaways = await cur.fetchall()

                for g in giveaways:

                    gid,msg,channel,winners,prize = g

                    c = self.bot.get_channel(channel)

                    if c is None:
                        continue

                    users = await db.execute_fetchall("""
                    SELECT user_id
                    FROM giveaway_entries
                    WHERE giveaway_id=?
                    """,(gid,))

                    users=[u[0] for u in users]

                    if len(users)==0:

                        winner="Niemand"

                    else:

                        winner=random.sample(
                            users,
                            min(
                                winners,
                                len(users)
                            )
                        )

                        winner=", ".join(
                            f"<@{u}>"
                            for u in winner
                        )

                    message=await c.fetch_message(msg)

                    embed=message.embeds[0]

                    embed.color=discord.Color.red()

                    embed.add_field(
                        name="Gewinner",
                        value=winner,
                        inline=False
                    )

                    await message.edit(
                        embed=embed,
                        view=None
                    )

                    await c.send(
                        f"🎉 Herzlichen Glückwunsch {winner}"
                    )

                    await db.execute("""
                    UPDATE giveaways
                    SET ended=1
                    WHERE id=?
                    """,(gid,))

                await db.commit()

            await asyncio.sleep(15)

    @app_commands.command(
        name="giveaway",
        description="Giveaway starten"
    )
    @app_commands.describe(
        prize="Preis",
        minutes="Laufzeit",
        winners="Gewinner"
    )
    async def giveaway(
        self,
        interaction: discord.Interaction,
        prize:str,
        minutes:int,
        winners:int
    ):

        end=int(time.time())+(minutes*60)

        embed=discord.Embed(
            title="🎉 Giveaway",
            description=f"Preis: **{prize}**",
            color=discord.Color.gold()
        )

        embed.add_field(
            name="Gewinner",
            value=str(winners)
        )

        embed.add_field(
            name="Ende",
            value=f"<t:{end}:R>"
        )

        async with aiosqlite.connect(DB) as db:

            cur=await db.execute("""
            INSERT INTO giveaways(
                guild_id,
                channel_id,
                message_id,
                prize,
                winners,
                host,
                end_time
            )
            VALUES(
                ?,?,?,?,?,?,?
            )
            """,(
                interaction.guild.id,
                interaction.channel.id,
                0,
                prize,
                winners,
                interaction.user.id,
                end
            ))

            gid=cur.lastrowid

            view=GiveawayButton(gid)

            msg=await interaction.channel.send(
                embed=embed,
                view=view
            )

            await db.execute("""
            UPDATE giveaways
            SET message_id=?
            WHERE id=?
            """,(msg.id,gid))

            await db.commit()

        await interaction.response.send_message(
            "✅ Giveaway erstellt.",
            ephemeral=True
        )


async def setup(bot):
    await bot.add_cog(Giveaway(bot))
